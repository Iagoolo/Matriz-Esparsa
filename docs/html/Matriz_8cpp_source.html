Matriz Esparsa 2.0
O projeto visa otimizar o armazenamento e as operações em matrizes esparsas, que possuem maioria de elementos zero. Armazenando apenas os valores não nulos, economiza-se memória e acelera-se operações como soma, multiplicação e inversão. Um exemplo prático é o uso de matrizes em programas de planilha como o Excel.

120void Matriz::limpar()
121{
122 if (cabecalho == nullptr)
123 return;
124
125 // Obter o primeiro sentinela de linha.
126 Node *LinhaAtual = cabecalho->abaixo;
127 // Se não houver linhas (ou se o único nó for o cabeçalho), nada a fazer.
128 if (LinhaAtual == cabecalho)
129 return;
130
131 // Quebrar a circularidade vertical: encontrar a última linha.
132 Node *ColunaAtual = LinhaAtual;
133 while (ColunaAtual->abaixo != cabecalho)
134 {
135 ColunaAtual = ColunaAtual->abaixo;
136 }
137 // Quebramos a circularidade: a última linha passa a apontar para nullptr.
138 ColunaAtual->abaixo = nullptr;
139
140 // Percorrer cada sentinela de linha (a cadeia agora é linear)
141 for (Node *linha = LinhaAtual; linha != nullptr; linha = linha->abaixo)
142 {
143
144 // A lista horizontal de cada linha é circular, onde os nós de dados estão
145 // entre 'linha->direita' e o próprio sentinela (linha).
146 Node *cur = linha->direita;
147 while (cur != linha)
148 {
149 Node *proximo = cur->direita; // Guarda o próximo nó antes de deletar
150 delete cur; // Libera o nó de dado
151 cur = proximo;
152 }
153 // Restaura o ponteiro horizontal do sentinela para ele mesmo.
154 linha->direita = linha;
155 }
156
157 // Restaura a circularidade vertical: o último sentinela volta a apontar para o cabeçalho.
158 ColunaAtual = LinhaAtual;
159 while (ColunaAtual->abaixo != nullptr)
160 {
161 ColunaAtual = ColunaAtual->abaixo;
162 }
163 ColunaAtual->abaixo = cabecalho;
164}

166void Matriz::insert(const int &posI, const int &posJ, const double &value)
167{
168 // Não armazena valores iguais a zero
169 if (value == 0)
170 return;
171
172 // Validação dos índices
173 if (posI <= 0 || posI > linhas || posJ <= 0 || posJ > colunas)
174 throw std::invalid_argument("Erro: Local de inserção inválido");
175
176 // Busca na lista horizontal (linha)dede
177 Node *linhaAtual = cabecalho;
178 while (linhaAtual->linha < posI)
179 {
180 linhaAtual = linhaAtual->abaixo;
181 }
182
183 // Procura pela posição correta na linha
184 Node *aux = linhaAtual;
185 while (aux->direita != linhaAtual && aux->direita->coluna < posJ)
186 {
187 aux = aux->direita;
188 }
189
190 // Verifica se o nó já existe
191 if (aux->direita->coluna == posJ)
192 {
193 aux->direita->atualizaValor(value); // Atualiza o valor
194 return;
195 }
196
197 // Cria um novo nó
198 Node *novo = new Node(posI, posJ, value);
199
200 // Insere o nó na lista horizontal
201 novo->direita = aux->direita;
202 aux->direita = novo;
203
204 // Busca na lista vertical (coluna)
205 Node *colunaAtual = cabecalho;
206 while (colunaAtual->coluna < posJ)
207 {
208 colunaAtual = colunaAtual->direita;
209 }
210
211 // Procura pela posição correta na coluna
212 aux = colunaAtual;
213 while (aux->abaixo != colunaAtual && aux->abaixo->linha < posI)
214 {
215 aux = aux->abaixo;
216 }
217
218 // Insere o nó na lista vertical
219 novo->abaixo = aux->abaixo;
220 aux->abaixo = novo;
